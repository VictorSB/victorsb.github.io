Custom elements

Custom elements

  • We are going to create a new HTML tag.
  • We will create a custom element prototype extending HTMLElement.prototype
  • It defines lifecycle callbacks.
  • Could have methods, getters, setters its own style...

Custom elements


//Create and prepare the prototype
var helloPrototype = Object.create(HTMLElement.prototype);
helloPrototype.createdCallback = function() { 
	this.innerHTML = "
HelloWorld
";
};

helloPrototype.hello = function(){
	console.log("hello");	
};

Custom elements


//Register the custom element
document.registerElement('hello-element', {
	prototype: helloPrototype,
});

Custom elements


//Creating an instance of our element
var helloInstance = document.createElement('hello-element');

//Appending the element
document.body.appendChild(helloInstance);

//Using the element
helloInstance.hello();

  • polymer-mini: +DOM control, template stamping
  • polymer: +data binding, closest to 0.5

Polymer({
	is: 'hello-component',

	properties: {
	  author: {
	    type: String,
	    value: 'mmtv'
	  },

	  fancy: {
	    type: Boolean,
	    type: false
	  }

	},

	ready: function() {
	  console.log('ready');
	},

	sayHello: function(greeting) {
	  var response = greeting || 'Hello World!';
	  return 'hello-component says, ' + response;
	},

	fireLasers: function() {
	  this.fire('hello-component-lasers-success',
	  { sound: 'Pew pew pew!' });
	}
});
