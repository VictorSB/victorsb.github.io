IndexedDB & Samples (I)

Opening the database...

var myindexeddb = {}
myindexeddb.db = null;

myindexeddb.open = function() {
  var request = webkitIndexedDB.open("samples");

  request.onsuccess = function(e) {
    myindexeddb = e.target.result;
    // Do some more stuff in a minute
  };

  request.onfailure = myindexeddb.onerror;
};

IndexedDB & Samples (II)

Creating an object store...

myindexeddb.open = function() {
  var request = webkitIndexedDB.open("samples",
    "This is a description of the database.");

  request.onsuccess = function(e) {
    var v = "1.0";
    myindexeddb.db = e.target.result;
    var db = myindexeddb.db;
    // We can only create Object stores in a setVersion transaction;
    if(v!= db.version) {
      var setVrequest = db.setVersion(v);

      // onsuccess is the only place we can create Object Stores
      setVrequest.onfailure = myindexeddb.onerror;
      setVrequest.onsuccess = function(e) {
        var store = db.createObjectStore("samples",
          {keyPath: "timeStamp"});

        myindexeddb.getAllSamples();
      };
    }
    myindexeddb.getAllSamples();
  };
  request.onfailure = myindexeddb.onerror;
}

IndexedDB & Samples (II)

Adding an object...

myindexeddb.addSample = function(text) {
  var db = myindexeddb.db;
  var trans = db.transaction(["sample"], webkitIDBTransaction.READ_WRITE, 0);
  var store = trans.objectStore("sample");
  var request = store.put({
    "text": text,
    "timeStamp" : new Date().getTime()
  });
};
